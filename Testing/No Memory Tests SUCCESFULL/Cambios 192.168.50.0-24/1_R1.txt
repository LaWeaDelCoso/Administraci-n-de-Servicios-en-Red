! ============================================================
! Reconfiguracion de R1 ante subred(es) sorpresa
! Modo de estado: NO MEMORY (estado base al cierre de pruebas)
! ============================================================
enable
configure terminal
!
! --- FastEthernet1/0 (hacia LAN1 / Switch1) ---
interface FastEthernet1/0
 no ip address
 ip address 192.168.50.1 255.255.255.0
 description SUBRED_SORPRESA_LAN1_APLICADA
!
router ospf 1
 no network 192.168.20.0 0.0.0.255 area 0
 network 192.168.50.0 0.0.0.255 area 0
end
write memory

! --- Verificacion sugerida ---
! show ip interface brief
! show ip ospf neighbor
! show ip route ospf
