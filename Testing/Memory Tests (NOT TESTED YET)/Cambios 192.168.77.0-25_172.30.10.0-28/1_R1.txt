configure terminal
!

 no ip address

 description SUBRED_SORPRESA_LAN1_APLICADA

 ip helper-address 172.30.10.2
!
! --- FastEthernet2/0 (hacia LAN2 / Switch2) ---
interface FastEthernet2/0
 no ip address
 ip address 172.30.10.1 255.255.255.240

 no network 192.168.50.0 0.0.0.255 area 0

 network 172.30.10.0 0.0.0.15 area 0
end
